\question What is the capital of France?
\begin{choices}
  \choice Berlin
  \choice Madrid
  \choice Paris
  \choice Rome
\end{choices}

\question Which of the following is equal to \( 2^3 \)?
\begin{choices}
  \choice 4
  \choice 6
  \choice 8
  \choice 16
\end{choices}

\question The derivative of \( f(x) = x^2 \) with respect to \( x \) is:
\begin{choices}
  \choice \(2x\)
  \choice \(x\)
  \choice \(x^2\)
  \choice \(2\)
\end{choices}

\question Which G-code is commonly used for linear interpolation with feed rate?
\begin{choices}
  \choice G00
  \choice G01
  \choice G02
  \choice G90
\end{choices}

\question A multiple-choice test is best stored in a database as:
\begin{choices}
  \choice Images only
  \choice Plain text with choices and correct answer
  \choice Audio files
  \choice Video lectures
\end{choices}

\end{questions}
